void print(vector<vector<int>>& adj)
{
	for (int i = 0; i < adj.size(); ++i)
	{
		cout << setw(2) << i << " : ";
		for (int x : adj[i])
			cout << setw(2) << x;
		cout << endl;
	}
	cout << endl;
}

void print(vector<int>& v)
{
	for (int x : v)
		cout << setw(2) << x;
	cout << endl;
}
