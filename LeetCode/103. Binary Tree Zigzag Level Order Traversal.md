# Medium

Given the $root$ of a binary tree, return the zigzag level order traversal of its nodes' values.

```cpp
class Solution {
public:
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
        vector<vector<int>> ret;
        deque<TreeNode*> dq;
        int lvl = 0;
        int s = 0;
        TreeNode *cur = NULL;
        
        if (root)
            dq.push_back(root);
        
        while (s = dq.size())
        {
            vector<int> tmp;
            
            if (lvl % 2)
            {
                while (-- s >= 0)
                {
                    cur = dq.back();
                    dq.pop_back();
                    
                    tmp.push_back(cur->val);
                    
                    if (cur->right)
                        dq.push_front(cur->right);
                    
                    if (cur->left)
                        dq.push_front(cur->left);
                }
            }
            else
            {
                while (-- s >= 0)
                {
                    cur = dq.front();
                    dq.pop_front();
                    
                    tmp.push_back(cur->val);
                    
                    if (cur->left)
                        dq.push_back(cur->left);
                    
                    if (cur->right)
                        dq.push_back(cur->right);
                }
            }
            
            ret.push_back(tmp);
            lvl = 1 - lvl;
        }
        
        return ret;
    }
};
```
